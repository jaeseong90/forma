FORMA Component API Reference

좌측 메뉴에서 컴포넌트를 선택하세요.

화면 기본 구조 (IIFE 패턴)

(function() {
    var PGM = 'SDA010';
    var listener = platform.initListener(PGM);
    var ctx = {};

    listener.initPgm = function() {
        ctx.searchForm = new FormaForm('#search-area', { search: true, elements: [...] });
        ctx.grid1 = new FormaGrid('#grid-area', { editable: true, columns: [...] });
        listener.button.search.click();
    };

    listener.button.search.click = function() {
        platform.post('/sda010/selectGrid1', ctx.searchForm.getData(), new Callback(function(result) {
            if (result.resultCode === RESULT_CODE.OK) ctx.grid1.setData(result.resultData);
        }));
    };

    document.addEventListener('DOMContentLoaded', () => platform.startPage(PGM, listener));
})();

FormaGrid

ERP 수준의 데이터 그리드. 인라인 편집, 셀 병합, 트리, 가상 스크롤, 그룹핑, 키보드, 클립보드, Undo/Redo.

생성

var grid = new FormaGrid('#grid-area', {
    editable: true,       // 인라인 편집
    checkable: true,      // 체크박스 컬럼
    sortable: true,       // 헤더 클릭 정렬
    rowNum: true,          // 행번호 컬럼
    paging: true,          // 페이징 바
    pageSize: 50,          // 페이지 크기
    columns: [...]         // 컬럼 정의
});

FormaForm

21종 위젯을 지원하는 폼 컴포넌트. 검색폼/입력폼 모드, 변경 감지(isDirty), 검증(validate).

생성

// 검색폼 (가로 배치)
var searchForm = new FormaForm('#search-area', {
    search: true,
    elements: [
        { field: 'keyword', label: '검색어', widget: 'text' },
        { field: 'date', label: '기간', widget: 'dateRange', default: 'THIS_MONTH' }
    ]
});

// 입력폼 (세로 배치)
var form = new FormaForm('#form-area', {
    columns: 2, labelWidth: 100,
    elements: [
        { field: 'name', label: '이름', required: true },
        { field: 'dept', label: '부서', widget: 'combo', code: 'DEPT_TYPE' }
    ]
});

위젯 타입

widget설명주요 속성
text텍스트 입력 (기본)placeholder, maxLength, type='number'
number숫자 입력min, max, step, suffix, prefix
combo드롭다운options: [{value, label}], code: 'GROUP'
date날짜 선택readOnly
dateRange기간 (from~to)default: 'THIS_MONTH', 'TODAY', 'THIS_YEAR'
codePopup코드 팝업popup: {url, codeField, nameField, width, height}
textarea텍스트영역rows
checkbox체크박스checkLabel
radio라디오 버튼options: [{value, label}]
hidden숨김 필드—
currency금액 입력콤마 자동포맷, getData()에서 Number
multiCombo복수 선택getData()에서 쉼표구분 문자열
yearMonth년월 선택default: 'THIS_MONTH'
year년도 선택default: 'THIS_YEAR', range: 5
switchON/OFF 토글onLabel, offLabel, getData()에서 'Y'/'N'
display읽기전용 텍스트/뱃지badge: true, badgeColors: {값: 색상}
password비밀번호눈 아이콘 토글
divider구분선/섹션label (텍스트)

공통 필드 속성

속성타입설명
fieldString데이터 필드명
labelString라벨 텍스트
requiredBoolean필수 여부 (라벨에 * 표시)
readOnlyBoolean읽기 전용
colspanNumber차지할 컬럼 수 (form 모드)

suffixString입력 뒤 표시 ('원', 'kg', '%')
helpTextString필드 아래 도움말
onChangeFunction(value, oldValue)
dependsOnObject의존 콤보 {field, paramKey}

API 메서드

메서드설명

setField(field, value)개별 필드 값 세팅
clear()전체 초기화
validate()required 필드 검증

setOptions(field, options)콤보 옵션 교체
showField(field) / hideField(field)필드 표시/숨김

FormaModal

팝업 모달. 중첩 지원, 드래그 이동, ESC 닫기.

사용법

var modal = new FormaModal({
    title: '거래처 검색',

        console.log(result.CUST_CD, result.CUST_NM);

modal.show(param);  // param은 팝업에서 FormaModal.getParam()으로 수신

YAML Runtime Engine

YAML 설계서 1파일로 완전한 CRUD 화면을 런타임에 생성합니다.

YAML 파일 위치

design/screens/{SCREEN_ID}.yml

화면 타입

typeDOM 구조동작
list검색폼 + grid1기본 CRUD
split-detail검색폼 + [grid1 | grid2]grid1 클릭 → grid2 로드
master-detail검색폼 + grid1 + form + grid2grid1 클릭 → form + grid2
split-tab검색폼 + [grid1 | tab(form+grids)]복합 화면

API 엔드포인트

GET /api/screen/{id}/definition화면 정의 조회
POST /api/screen/{id}/selectGrid1Grid1 데이터 조회
POST /api/screen/{id}/selectGrid2Grid2 데이터 조회
POST /api/screen/{id}/saveGrid1Grid1 저장 (gstat 기반 I/U)
POST /api/screen/{id}/saveGrid2Grid2 저장
POST /api/screen/{id}/deleteGrid1Grid1 삭제
GET /api/screen/_reloadYAML 재로딩

YAML 전체 구조

screen:
  id: SDA010
  name: 거래처관리
  type: list              # list | split-detail | master-detail | split-tab

auth:
  search: true
  new: true
  save: true
  delete: true

search:
  - { field: keyword, label: 검색어, widget: text }
  - { field: type, label: 구분, widget: combo, code: CUST_TYPE }

grids:
  grid1:
    editable: true
    checkable: true
    paging: true
    pageSize: 50
    columns:
      - { field: CUST_CD, label: 거래처코드, width: 100 }
      - { field: CUST_NM, label: 거래처명, width: 200, editor: text, required: true }

form:                     # master-detail, split-tab에서 사용
  masterForm:
    columns: 2
    elements:
      - { field: ORDER_NO, label: 수주번호, readOnly: true }

sql:
  tables:
    - { name: tb_customer }
  selectGrid1:
    table: tb_customer
    columns: cust_cd, cust_nm, cust_type
    orderBy: cust_cd
  selectGrid2:
    table: tb_order_item
    where: "order_no = #{order_no}"
